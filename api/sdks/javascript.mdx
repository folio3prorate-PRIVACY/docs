## Quick Start

<Steps>
  <Step title="Get your API Key" icon="key">

    First, obtain your API key from the [Octav Developer Portal](https://data.octav.fi/).

    <Warning>
      **Never hardcode your API key!** Always use environment variables or a secure configuration management system.
    </Warning>

  </Step>

  <Step title="Initialize the Client" icon="code">

    <Tabs>
      <Tab title="JavaScript">
        ```javascript
        import { OctavClient } from '@octav/sdk';

        const octav = new OctavClient({
          apiKey: process.env.OCTAV_API_KEY
        });
        ```
      </Tab>

      <Tab title="TypeScript">
        ```typescript
        import { OctavClient } from '@octav/sdk';

        const octav = new OctavClient({
          apiKey: process.env.OCTAV_API_KEY!
        });
        ```
      </Tab>

      <Tab title="CommonJS">
        ```javascript
        const { OctavClient } = require('@octav/sdk');

        const octav = new OctavClient({
          apiKey: process.env.OCTAV_API_KEY
        });
        ```
      </Tab>
    </Tabs>

  </Step>

  <Step title="Make Your First Request" icon="paper-plane">

    ```javascript
    // Get portfolio data
    const portfolios = await octav.portfolios.list();
    console.log(portfolios);

    // Get wallet information
    const wallet = await octav.wallets.get('0xabcdef1234567890');
    console.log(`Total Value: $${wallet.total_value}`);
    ```

  </Step>
</Steps>

## API Reference

<Tabs>
  <Tab title="Portfolios" icon="wallet">

    ### List All Portfolios

    ```javascript
    const portfolios = await octav.portfolios.list();
    ```

    ### Get Portfolio Details

    ```javascript
    const portfolio = await octav.portfolios.get('portfolio_123');
    ```

    **Response:**
    ```json
    {
      "id": "portfolio_123",
      "name": "My Portfolio",
      "total_value": 15231.89,
      "24h_change": 0.28,
      "assets": [...],
      "wallets": [...]
    }
    ```

    ### Create Portfolio

    ```javascript
    const newPortfolio = await octav.portfolios.create({
      name: 'My New Portfolio',
      description: 'Long-term holdings',
      wallets: ['0xabc...', '0xdef...']
    ```

    ```json

      "current_value": 15231.89,
      "start_value": 14850.00,
      "change_percent": 2.57,
      "historical_values": [...]
    ```

    ```typescript
    // Type-safe responses
    const portfolios: Portfolio[] = await octav.portfolios.list();
    const wallet: Wallet = await octav.wallets.get('0xabc...');
    const transactions: Transaction[] = await octav.transactions.list('0xabc...');
    ```

  </Tab>

  <Tab title="Custom Types">

## Error Handling

The SDK throws typed errors that you can catch and handle:

<Tabs>
  <Tab title="Basic Error Handling">
    ```javascript
    import { OctavClient, OctavAPIError } from '@octav/sdk';

    try {
      const wallet = await octav.wallets.get('invalid_address');
    } catch (error) {
      if (error instanceof OctavAPIError) {
        console.error(`API Error: ${error.message}`);
        console.error(`Status Code: ${error.statusCode}`);
      } else {
        console.error('Unexpected error:', error);
      }
    }
    ```
  </Tab>

  <Tab title="Specific Error Types">
    ```javascript
    import {
      OctavAPIError,
      RateLimitError,
      AuthenticationError,
      NotFoundError
    } from '@octav/sdk';

    try {
      const wallet = await octav.wallets.get('0xabc...');
    } catch (error) {
      if (error instanceof AuthenticationError) {
        console.error('Invalid API key');
      } else if (error instanceof RateLimitError) {
        console.error('Rate limit exceeded, retry after:', error.retryAfter);
      } else if (error instanceof NotFoundError) {
        console.error('Wallet not found');
      } else if (error instanceof OctavAPIError) {
        console.error(`API Error ${error.statusCode}:`, error.message);
      }
    }
    ```
  </Tab>

  <Tab title="With Retry Logic">
    ```javascript
    async function getWalletWithRetry(address, maxRetries = 3) {
      for (let i = 0; i < maxRetries; i++) {
        try {
          return await octav.wallets.get(address);
        } catch (error) {
          if (error instanceof RateLimitError && i < maxRetries - 1) {
            // Wait before retrying
            await new Promise(resolve => setTimeout(resolve, error.retryAfter * 1000));
            continue;
          }
          throw error;
        }
      }
    }
    ```
  </Tab>
</Tabs>

## Pagination

Handle paginated responses efficiently:

<Tabs>
  <Tab title="Manual Pagination">
    ```javascript
    let allTransactions = [];
    let offset = 0;
    const limit = 100;

    while (true) {
      const response = await octav.transactions.list('0xabc...', {
        limit,
        offset
      });

      allTransactions.push(...response.data.transactions);

      if (response.data.transactions.length < limit) {
        break; // No more data
      }

      offset += limit;
    }

    console.log(`Fetched ${allTransactions.length} transactions`);
    ```
  </Tab>

  <Tab title="Async Iterator">
    ```javascript
    // Using async iterator (if supported by SDK)
    for await (const transaction of octav.transactions.iterate('0xabc...')) {
      console.log(transaction);

      // Process each transaction
      if (transaction.value > 1000) {
        await processHighValueTransaction(transaction);
      }
    }
    ```
  </Tab>

  <Tab title="Helper Function">
    ```javascript
    async function* paginateAll(fetchFn, options = {}) {
      const limit = options.limit || 100;
      let offset = 0;

      while (true) {
        const response = await fetchFn({ ...options, limit, offset });

        for (const item of response.data) {
          yield item;
        }

        if (response.data.length < limit) break;
        offset += limit;
      }
    }

    // Usage
    for await (const wallet of paginateAll(
      (opts) => octav.wallets.list(opts)
    )) {
      console.log(wallet);
    }
    ```
  </Tab>
</Tabs>

## Real-World Examples

<AccordionGroup>

  <Accordion title="Track Multiple Wallets" icon="users">

    ```javascript
    const walletAddresses = [
      '0xabc...',
      '0xdef...',
      '0x123...'
    ];

    // Fetch all wallets in parallel
    const walletData = await Promise.all(
      walletAddresses.map(addr => octav.wallets.get(addr))
    );

    // Calculate total value
    const totalValue = walletData.reduce(
      (sum, wallet) => sum + wallet.total_value,
      0
    );

    console.log(`Total Portfolio Value: $${totalValue.toFixed(2)}`);

    // Group by chain
    const byChain = walletData.flatMap(w => w.assets)
      .reduce((acc, asset) => {
        acc[asset.chain] = (acc[asset.chain] || 0) + asset.value;
        return acc;
      }, {});

    console.log('Value by chain:', byChain);
    ```

    ```javascript
    import { createObjectCsvWriter } from 'csv-writer';

    async function exportTransactionsToCsv(walletAddress, filename) {
      // Fetch all transactions
      const allTransactions = [];
      let offset = 0;
      const limit = 100;

      while (true) {
        const response = await octav.transactions.list(walletAddress, {
          limit,
          offset
        });

        allTransactions.push(...response.data.transactions);

        if (response.data.transactions.length < limit) break;
        offset += limit;
      }

      // Write to CSV
      const csvWriter = createObjectCsvWriter({
        path: filename,
        header: [
          { id: 'hash', title: 'Transaction Hash' },
          { id: 'type', title: 'Type' },
          { id: 'timestamp', title: 'Date' },
          { id: 'value', title: 'Value (USD)' },
          { id: 'chain', title: 'Chain' }
        ]
      });

      await csvWriter.writeRecords(allTransactions);
      console.log(`Exported ${allTransactions.length} transactions to ${filename}`);
    }

    await exportTransactionsToCsv('0xabc...', 'transactions.csv');
    ```
