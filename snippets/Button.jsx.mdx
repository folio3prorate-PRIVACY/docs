export const Button = ({ href, children, variant = "primary" }) => {
  const variants = {
    primary: "bg-[#3F58C7] hover:bg-[#2F4AA7] text-white",
    secondary: "bg-[#6BCCD1] hover:bg-[#5ABCC1] text-white",
    outline: "border-2 border-[#3F58C7] text-[#3F58C7] hover:bg-[#3F58C7] hover:text-white"
  };

  return (
    <a href={href} className={`inline-block rounded-lg px-6 py-3 font-medium transition-all duration-200 ${variants[variant]}`}>
      {children}
    </a>
  );
};
